async function loadCurve() { const deviceId = document.getElementById("deviceSelect").value; const status = document.getElementById("status"); status.textContent = "⏳ Chargement..."; const mesures = await fetch( `${SUPABASE_URL}/rest/v1/mesures?device_id=eq.${deviceId}&order=timestamp.asc`, { headers: { "apikey": SUPABASE_KEY, "Authorization": "Bearer " + SUPABASE_KEY } } ).then(r => r.json()); if (!mesures.length) { status.textContent = "Aucune mesure."; return; } const points = mesures.map(m => ({ x: new Date(m.timestamp), y: m.humidity })); if (chart) chart.destroy(); chart = new Chart(document.getElementById("chart"), { type: "line", data: { datasets: [{ label: "Humidité (%)", data: points, borderColor: "#2c7", tension: 0.2 }] }, options: { responsive: true, scales: { x: { type: "time" }, y: { min: 0, max: 100 } } } }); status.textContent = "✅ Courbe affichée"; }
